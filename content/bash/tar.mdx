import { CommandSession, Command, CommandOutput, Flag, Gloss, Kw, FileRef,
  SourceAnnot,
} from '~components/commands'

### Expand a `.tar.gz` file

<CommandSession>
<Command>ls</Command>
<CommandOutput>fruits.tar.gz</CommandOutput>
<Command>tar -<Flag 
  short='x' long='extract' gloss='extract files from an archive' kw={1}
  >x</Flag><Flag
  short='f' long='file' 
  kw={2}
  >f</Flag> fruits.tar.gz</Command>
<Command>ls</Command>
<CommandOutput>apple.png banana.png canteloupe.png fruits.tar.gz</CommandOutput>
</CommandSession>

<Gloss>
<Kw kw={1}>Extracts</Kw> the contents of the <Kw kw={2}>file</Kw> <FileRef>fruits.tar.gz</FileRef> into the current directory.
</Gloss>

### Expand a `.tar` file

<CommandSession>
<Command>tar -xf fruits.tar</Command>
</CommandSession>

### Archive a directory

<CommandSession>
<Command>tar -<Flag
  kw={2} short='c' long='create' gloss='create a new archive'
>c</Flag><Flag
  kw={3} short='z' long='gzip' gloss='use gzip compression'
>z</Flag>f fruits.tar.gz fruits/</Command>
</CommandSession>

<Kw kw={2}>Create</Kw> the file <code>fruits.tar.gz</code> by <Kw kw={3}>compressing</Kw> the directory  <code>fruits/</code>.

### Archive specific files

<CommandSession>
<Command>tar -cxf fruits.tar.gz apple.png banana.png canteloupe.png</Command>
</CommandSession>

Creates a compressed tarball `fruits.tar.gz` containing the three specified image files.

### List the contents of a tarball

<CommandSession>
<Command>tar -<Flag
  kw={1} short='t' long='list' gloss='list the contents of an archive'
>t</Flag>f fruits.tar.gz</Command>
<CommandOutput>{`apple.png
banana.png
canteloupe.png`}
</CommandOutput>
</CommandSession>
